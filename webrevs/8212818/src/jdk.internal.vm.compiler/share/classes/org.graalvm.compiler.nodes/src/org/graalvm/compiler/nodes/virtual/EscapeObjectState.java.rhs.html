  25 package org.graalvm.compiler.nodes.virtual;
  26 
  27 import static org.graalvm.compiler.nodeinfo.NodeCycles.CYCLES_0;
  28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_0;
  29 
  30 import org.graalvm.compiler.graph.Node.ValueNumberable;
  31 import org.graalvm.compiler.graph.NodeClass;
  32 import org.graalvm.compiler.nodeinfo.NodeInfo;
  33 import org.graalvm.compiler.nodes.VirtualState;
  34 
  35 @NodeInfo(cycles = CYCLES_0, size = SIZE_0)
  36 public abstract class EscapeObjectState extends VirtualState implements ValueNumberable {
  37     public static final NodeClass<EscapeObjectState> TYPE = NodeClass.create(EscapeObjectState.class);
  38 
  39     @OptionalInput protected VirtualObjectNode object;
  40 
  41     public VirtualObjectNode object() {
  42         return object;
  43     }
  44 
  45     public EscapeObjectState(NodeClass<? extends EscapeObjectState> c, VirtualObjectNode object) {
  46         super(c);
  47         this.object = object;
  48     }
  49 
  50     @Override
  51     public abstract EscapeObjectState duplicateWithVirtualState();
  52 
  53     @Override
  54     public boolean isPartOfThisState(VirtualState state) {
  55         return this == state;
  56     }
  57 
  58     @Override
  59     public void applyToVirtual(VirtualClosure closure) {
  60         closure.apply(this);
  61     }
  62 }
