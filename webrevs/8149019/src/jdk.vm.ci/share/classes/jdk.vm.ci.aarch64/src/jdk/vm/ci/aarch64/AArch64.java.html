 147     /**
 148      * Basic set of CPU features mirroring what is returned from the cpuid instruction. See:
 149      * {@code VM_Version::cpuFeatureFlags}.
 150      */
 151     public enum CPUFeature {
 152         FP,
 153         ASIMD,
 154         EVTSTRM,
 155         AES,
 156         PMULL,
 157         SHA1,
 158         SHA2,
 159         CRC32,
 160         A53MAC,
 161         DMB_ATOMICS
 162     }
 163 
 164     private final EnumSet<CPUFeature> features;
 165 
 166     /**
 167      * Set of flags to control code emission.
 168      */
 169     public enum Flag {
 170         UseBarriersForVolatile,
 171         UseCRC32,
 172         UseNeon
 173     }
 174 
 175     private final EnumSet<Flag> flags;
 176 
 177     public AArch64(EnumSet<CPUFeature> features, EnumSet<Flag> flags) {
 178         super("aarch64", AArch64Kind.QWORD, ByteOrder.LITTLE_ENDIAN, true, allRegisters, 0, 0, 0);
 179         this.features = features;
 180         this.flags = flags;
 181     }
 182 
 183     public EnumSet<CPUFeature> getFeatures() {
 184         return features;
 185     }
 186 
 187     public EnumSet<Flag> getFlags() {
 188         return flags;
 189     }
 190 
 191     @Override
 192     public PlatformKind getPlatformKind(JavaKind javaKind) {
 193         switch (javaKind) {
 194             case Boolean:
 195             case Byte:
 196                 return AArch64Kind.BYTE;
 197             case Short:
 198             case Char:
 199                 return AArch64Kind.WORD;
 200             case Int:
 201                 return AArch64Kind.DWORD;
 202             case Long:
 203             case Object:
 204                 return AArch64Kind.QWORD;
 205             case Float:
 206                 return AArch64Kind.SINGLE;
 207             case Double:
 208                 return AArch64Kind.DOUBLE;
 209             default:
 210                 return null;
 211         }
 212     }
 213 
 214     @Override
 215     public boolean canStoreValue(RegisterCategory category, PlatformKind platformKind) {
 216         AArch64Kind kind = (AArch64Kind) platformKind;
 217         if (kind.isInteger()) {
 218             return category.equals(CPU);
 219         } else if (kind.isSIMD()) {
 220             return category.equals(SIMD);
 221         }
 222         return false;
 223     }
 224 
 225     @Override
 226     public AArch64Kind getLargestStorableKind(RegisterCategory category) {
 227         if (category.equals(CPU)) {
 228             return AArch64Kind.QWORD;
 229         } else if (category.equals(SIMD)) {
 230             return AArch64Kind.V128_QWORD;
 231         } else {
 232             return null;
 233         }
 234     }
