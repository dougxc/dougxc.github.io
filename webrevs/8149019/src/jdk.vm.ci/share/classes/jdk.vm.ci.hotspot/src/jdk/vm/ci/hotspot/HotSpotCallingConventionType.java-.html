  23 package jdk.vm.ci.hotspot;
  24 
  25 import jdk.vm.ci.code.CallingConvention;
  26 import jdk.vm.ci.code.CallingConvention.Type;
  27 
  28 public enum HotSpotCallingConventionType implements CallingConvention.Type {
  29     /**
  30      * A request for the outgoing argument locations at a call site to Java code.
  31      */
  32     JavaCall(true),
  33 
  34     /**
  35      * A request for the incoming argument locations.
  36      */
  37     JavaCallee(false),
  38 
  39     /**
  40      * A request for the outgoing argument locations at a call site to external native code that
  41      * complies with the platform ABI.
  42      */
  43     NativeCall(true);
  44 
  45     /**
  46      * Determines if this is a request for the outgoing argument locations at a call site.
  47      */
  48     public final boolean out;
  49 
  50     public static final Type[] VALUES = values();
  51 
  52     private HotSpotCallingConventionType(boolean out) {
  53         this.out = out;
  54     }
  55 }
