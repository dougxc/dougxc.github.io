  61 /**
  62  * Test for {@code jdk.vm.ci.hotspot.CompilerToVM.resolvePossiblyCachedConstantInPool} method
  63  */
  64 public class ResolvePossiblyCachedConstantInPoolTest {
  65 
  66     public static void main(String[] args) throws Exception {
  67         Map<ConstantTypes, Validator> typeTests = new HashMap<>();
  68         typeTests.put(CONSTANT_STRING, ResolvePossiblyCachedConstantInPoolTest::validateString);
  69         ConstantPoolTestCase testCase = new ConstantPoolTestCase(typeTests);
  70         // The next "Class.forName" is here for the following reason.
  71         // When class is initialized, constant pool cache is available.
  72         // This method works only with cached constant pool.
  73         for (DummyClasses dummy : DummyClasses.values()) {
  74             Class.forName(dummy.klass.getName());
  75         }
  76         testCase.test();
  77     }
  78 
  79     private static void validateString(ConstantPool constantPoolCTVM,
  80                                        ConstantTypes cpType,
  81                                        DummyClasses dummyClass,
  82                                        int cpi) {
  83         TestedCPEntry entry = cpType.getTestedCPEntry(dummyClass, cpi);
  84         if (entry == null) {
  85             return;
  86         }
  87         int index = cpi;
  88         String cached = "";
  89         int cpci = dummyClass.getCPCacheIndex(cpi);
  90         if (cpci != ConstantPoolTestsHelper.NO_CP_CACHE_PRESENT) {
  91             index = cpci;
  92             cached = "cached ";
  93         }
  94         Object constantInPool = CompilerToVMHelper.resolvePossiblyCachedConstantInPool(constantPoolCTVM, index);
  95         String stringToVerify = (String) constantInPool;
  96         String stringToRefer = entry.name;
  97         if (stringToRefer.equals("") && cpci != ConstantPoolTestsHelper.NO_CP_CACHE_PRESENT) {
  98             stringToRefer = null; // tested method returns null for cached empty strings
  99         }
 100         String msg = String.format("Wrong string accessed by %sconstant pool index %d", cached, index);
 101         Asserts.assertEQ(stringToRefer, stringToVerify, msg);
 102     }
 103 }
