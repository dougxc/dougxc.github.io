  61 import static compiler.jvmci.compilerToVM.ConstantPoolTestCase.ConstantTypes.CONSTANT_METHODHANDLE;
  62 import static compiler.jvmci.compilerToVM.ConstantPoolTestCase.ConstantTypes.CONSTANT_METHODTYPE;
  63 
  64 /**
  65  * Test for {@code jdk.vm.ci.hotspot.CompilerToVM.resolveConstantInPool} method
  66  */
  67 public class ResolveConstantInPoolTest {
  68 
  69     private static final String NOT_NULL_MSG
  70             = "Object returned by resolveConstantInPool method should not be null";
  71 
  72     public static void main(String[] args) throws Exception {
  73         Map<ConstantTypes, Validator> typeTests = new HashMap<>();
  74         typeTests.put(CONSTANT_METHODHANDLE, ResolveConstantInPoolTest::validateMethodHandle);
  75         typeTests.put(CONSTANT_METHODTYPE, ResolveConstantInPoolTest::validateMethodType);
  76         ConstantPoolTestCase testCase = new ConstantPoolTestCase(typeTests);
  77         testCase.test();
  78         // The next "Class.forName" and repeating "testCase.test()"
  79         // are here for the following reason.
  80         // The first test run is without dummy class initialization,
  81         // which means no constant pool cache exists.
  82         // The second run is with initialized class (with constant pool cache available).
  83         // Some CompilerToVM methods require different input
  84         // depending on whether CP cache exists or not.
  85         for (DummyClasses dummy : DummyClasses.values()) {
  86             Class.forName(dummy.klass.getName());
  87         }
  88         testCase.test();
  89     }
  90 
  91     private static void validateMethodHandle(ConstantPool constantPoolCTVM,
  92                                              ConstantTypes cpType,
  93                                              DummyClasses dummyClass,
  94                                              int index) {
  95         Object constantInPool = CompilerToVMHelper.resolveConstantInPool(constantPoolCTVM, index);
  96         String msg = String.format("%s for index %d", NOT_NULL_MSG, index);
  97         Asserts.assertNotNull(constantInPool, msg);
  98         if (!(constantInPool instanceof MethodHandle)) {
  99             msg = String.format("Wrong constant pool entry accessed by index"
 100                                         + " %d: %s, but should be subclass of %s",
 101                                 index,
 102                                 constantInPool.getClass(),
 103                                 MethodHandle.class.getName());
 104             throw new AssertionError(msg);
 105         }
 106     }
 107 
 108     private static void validateMethodType(ConstantPool constantPoolCTVM,
 109                                            ConstantTypes cpType,
 110                                            DummyClasses dummyClass,
 111                                            int index) {
 112         Object constantInPool = CompilerToVMHelper.resolveConstantInPool(constantPoolCTVM, index);
 113         String msg = String.format("%s for index %d", NOT_NULL_MSG, index);
 114         Asserts.assertNotNull(constantInPool, msg);
 115         Class mtToVerify = constantInPool.getClass();
 116         Class mtToRefer = MethodType.class;
 117         msg = String.format("Wrong method type class accessed by"
 118                                     + " constant pool index %d",
 119                             index);
 120         Asserts.assertEQ(mtToRefer, mtToVerify, msg);
 121     }
 122 }
