 156     /**
 157      * Basic set of CPU features mirroring what is returned from the cpuid instruction. See:
 158      * {@code VM_Version::cpuFeatureFlags}.
 159      */
 160     public enum CPUFeature {
 161         FP,
 162         ASIMD,
 163         EVTSTRM,
 164         AES,
 165         PMULL,
 166         SHA1,
 167         SHA2,
 168         CRC32,
 169         A53MAC,
 170         DMB_ATOMICS
 171     }
 172 
 173     private final EnumSet<CPUFeature> features;
 174 
 175     /**
 176      * Set of flags to control code emission.
 177      */
 178     public enum Flag {
 179         UseBarriersForVolatile,
 180         UseCRC32,
 181         UseNeon
 182     }
 183 
 184     private final EnumSet<Flag> flags;
 185 
 186     public AArch64(EnumSet<CPUFeature> features, EnumSet<Flag> flags) {
 187         super("aarch64", AArch64Kind.QWORD, ByteOrder.LITTLE_ENDIAN, true, allRegisters, 0, 0, 0);
 188         this.features = features;
 189         this.flags = flags;
 190     }
 191 
 192     public EnumSet<CPUFeature> getFeatures() {
 193         return features;
 194     }
 195 
 196     public EnumSet<Flag> getFlags() {
 197         return flags;
 198     }
 199 
 200     @Override
 201     public PlatformKind getPlatformKind(JavaKind javaKind) {
 202         switch (javaKind) {
 203             case Boolean:
 204             case Byte:
 205                 return AArch64Kind.BYTE;
 206             case Short:
 207             case Char:
 208                 return AArch64Kind.WORD;
 209             case Int:
 210                 return AArch64Kind.DWORD;
 211             case Long:
 212             case Object:
 213                 return AArch64Kind.QWORD;
 214             case Float:
 215                 return AArch64Kind.SINGLE;
 216             case Double:
 217                 return AArch64Kind.DOUBLE;
 218             default:
 219                 return null;
 220         }
 221     }
 222 
 223     @Override
 224     public boolean canStoreValue(RegisterCategory category, PlatformKind platformKind) {
 225         AArch64Kind kind = (AArch64Kind) platformKind;
 226         if (kind.isInteger()) {
 227             return category.equals(CPU);
 228         } else if (kind.isSIMD()) {
 229             return category.equals(SIMD);
 230         }
 231         return false;
 232     }
 233 
 234     @Override
 235     public AArch64Kind getLargestStorableKind(RegisterCategory category) {
 236         if (category.equals(CPU)) {
 237             return AArch64Kind.QWORD;
 238         } else if (category.equals(SIMD)) {
 239             return AArch64Kind.V128_QWORD;
 240         } else {
 241             return null;
 242         }
 243     }
