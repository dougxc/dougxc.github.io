  43 package com.amd.sumatra;
  44 
  45 import sun.misc.*;
  46 
  47 public class SumatraFactory {
  48 
  49     public static Sumatra getSumatra() {
  50         try {
  51             ClassLoader cl = Launcher.getLauncher().getClassLoader();
  52             Class c = Class.forName("com.oracle.graal.hotspot.hsail.ForEachToGraal", true, cl);
  53             Sumatra s = (Sumatra) c.newInstance();
  54             if (s != null) {
  55                 return s;
  56             } else {
  57                 System.out.println("### newInstance() is null?!");
  58                 return null;
  59             }
  60         } catch (ClassNotFoundException e) {
  61             System.out.println("Exception: " + e);
  62             e.printStackTrace();
  63             return null;
  64         } catch (IllegalAccessException e) {
  65             System.out.println("Exception: " + e);
  66             e.printStackTrace();
  67             return null;
  68         } catch (ExceptionInInitializerError e) {
  69             System.out.println("Exception: " + e);
  70             e.printStackTrace();
  71             return null;
  72         } catch (InstantiationException e) {
  73             System.out.println("Exception: " + e);
  74             e.printStackTrace();
  75             return null;
  76         }
  77     }
  78 }
