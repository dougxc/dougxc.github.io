1 package com.amd.sumatra;
   2 
   3 import java.lang.reflect.Field;
   4 
   5 public class SumatraUtils {
   6     public static Object getFieldFromObject(Field f, Object fromObj) {
   7         try {
   8             f.setAccessible(true);
   9             java.lang.reflect.Type type = f.getType();
  10 
  11             if (type == double.class) {
  12                 return f.getDouble(fromObj);
  13             } else if (type == float.class) {
  14                 return f.getFloat(fromObj);
  15             } else if (type == long.class) {
  16                 return f.getLong(fromObj);
  17             } else if (type == int.class) {
  18                 return f.getInt(fromObj);
  19             } else if (type == byte.class) {
  20                 return f.getByte(fromObj);
  21             } else if (type == boolean.class) {
  22                 return f.getBoolean(fromObj);
  23             } else {
  24                 // object
  25                 return f.get(fromObj);
  26             }
  27         } catch (Exception e) {
  28             // fail("unable to get field " + f + " from " + fromObj);
  29             return null;
  30         }
  31 
  32     }
  33 
  34 }
