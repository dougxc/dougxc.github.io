 168     public static int getStackOffset(Value reg) {
 169         return -(((StackSlot) reg).getRawOffset());
 170     }
 171 
 172     public static String mapStackSlot(Value reg) {
 173         StackSlot s = (StackSlot) reg;
 174         long offset = -s.getRawOffset();
 175         return "[%spillseg]" + "[" + offset + "]";
 176     }
 177 
 178     public static String mapRegister(Value arg) {
 179         return "$" + asRegister(arg).name;
 180     }
 181 
 182     @Override
 183     public boolean canStoreValue(RegisterCategory category, PlatformKind platformKind) {
 184         if (!(platformKind instanceof Kind)) {
 185             return false;
 186         }
 187         Kind kind = (Kind) platformKind;
 188         if (category == CPU) {
 189             switch (kind) {
 190                 case Boolean:
 191                 case Byte:
 192                 case Char:
 193                 case Short:
 194                 case Int:
 195                 case Long:
 196                 case Object:
 197                     return true;
 198             }
 199         } else if (category == FPU) {
 200             switch (kind) {
 201                 case Float:
 202                 case Double:
 203                     return true;
 204             }
 205         }
 206         return false;
 207     }
 208 
 209     @Override
 210     public PlatformKind getLargestStorableKind(RegisterCategory category) {
 211         if (category == CPU) {
 212             return Kind.Long;
 213         } else if (category == FPU) {
 214             return Kind.Double;
 215         } else {
 216             return Kind.Illegal;
 217         }
 218     }
