  39 package jdk.vm.ci.code.test;
  40 
  41 import org.junit.Assert;
  42 import org.junit.Test;
  43 
  44 import jdk.vm.ci.code.BytecodeFrame;
  45 import jdk.vm.ci.hotspot.HotSpotCodeCacheProvider;
  46 import jdk.vm.ci.meta.JavaKind;
  47 import jdk.vm.ci.meta.JavaValue;
  48 import jdk.vm.ci.meta.ResolvedJavaMethod;
  49 
  50 public class InterpreterFrameSizeTest extends CodeInstallationTest {
  51 
  52     HotSpotCodeCacheProvider hotspotCodeCache() {
  53         return (HotSpotCodeCacheProvider) codeCache;
  54     }
  55 
  56     @Test
  57     public void testNull() {
  58         try {
  59             hotspotCodeCache().interpreterFrameSize(null);
  60         } catch (NullPointerException npe) {
  61             // Threw NPE as expected.
  62             return;
  63         }
  64         Assert.fail("expected NullPointerException");
  65     }
  66 
  67     @Test
  68     public void test() {
  69         ResolvedJavaMethod resolvedMethod = metaAccess.lookupJavaMethod(getMethod("testNull"));
  70 
  71         int bci = 0;
  72         int numLocals = resolvedMethod.getMaxLocals();
  73         int numStack = 0;
  74         JavaValue[] values = new JavaValue[numLocals];
  75         JavaKind[] slotKinds = new JavaKind[numLocals];
  76         BytecodeFrame frame = new BytecodeFrame(null, resolvedMethod, bci, false, false, values, slotKinds, numLocals, numStack, 0);
  77         int size = hotspotCodeCache().interpreterFrameSize(frame);
  78         if (size <= 0) {
  79             Assert.fail("expected non-zero result");
  80         }
  81     }
  82 }
