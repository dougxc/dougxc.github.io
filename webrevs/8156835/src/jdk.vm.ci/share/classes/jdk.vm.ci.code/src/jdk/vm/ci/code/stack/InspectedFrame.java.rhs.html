  23 package jdk.vm.ci.code.stack;
  24 
  25 import jdk.vm.ci.meta.ResolvedJavaMethod;
  26 
  27 /**
  28  * Access to the object variables in a stack frame.
  29  */
  30 public interface InspectedFrame {
  31 
  32     /**
  33      * Returns the value of the object local at {@code index}. This value is a copy iff
  34      * {@link #isVirtual(int)} is true.
  35      */
  36     Object getLocal(int index);
  37 
  38     /**
  39      * Returns whether the local at {@code index} is a virtual object, and therefore the object
  40      * returned by {@link #getLocal(int)} is a copy.
  41      */
  42     boolean isVirtual(int index);
  43 
  44     /**
  45      * Returns true if the stack frame is a compiled stack frame and there are virtual objects
  46      * anywhere in the current state of the compiled method. This can return true even if
  47      * {@link #isVirtual(int)} return false for all locals.
  48      */
  49     boolean hasVirtualObjects();
  50 
  51     /**
  52      * This method will materialize all virtual objects, deoptimize the stack frame and make sure
  53      * that subsequent execution of the deoptimized frame uses the materialized values.
  54      */
  55     void materializeVirtualObjects(boolean invalidateCode);
  56 
  57     /**
  58      * @return the current bytecode index
  59      */
  60     int getBytecodeIndex();
  61 
  62     /**
  63      * @return the current method
  64      */
  65     ResolvedJavaMethod getMethod();
  66 
  67     /**
  68      * Checks if the current method is equal to the given method. This is semantically equivalent to
  69      * {@code method.equals(getMethod())}, but can be implemented more efficiently.
  70      */
  71     boolean isMethod(ResolvedJavaMethod method);
  72 }
