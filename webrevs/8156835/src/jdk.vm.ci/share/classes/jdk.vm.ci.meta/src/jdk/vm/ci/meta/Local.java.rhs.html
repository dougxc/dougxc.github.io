  23 package jdk.vm.ci.meta;
  24 
  25 public class Local {
  26 
  27     private final String name;
  28     private final int startBci;
  29     private final int endBci;
  30     private final int slot;
  31     private final JavaType type;
  32 
  33     public Local(String name, JavaType type, int startBci, int endBci, int slot) {
  34         this.name = name;
  35         this.startBci = startBci;
  36         this.endBci = endBci;
  37         this.slot = slot;
  38         this.type = type;
  39     }
  40 
  41     public int getStartBCI() {
  42         return startBci;
  43     }
  44 
  45     public int getEndBCI() {
  46         return endBci;
  47     }
  48 
  49     public String getName() {
  50         return name;
  51     }
  52 
  53     public JavaType getType() {
  54         return type;
  55     }
  56 
  57     public int getSlot() {
  58         return slot;
  59     }
  60 
  61     @Override
  62     public boolean equals(Object obj) {
  63         if (!(obj instanceof Local)) {
  64             return false;
  65         }
  66         Local that = (Local) obj;
  67         return this.name.equals(that.name) && this.startBci == that.startBci && this.endBci == that.endBci && this.slot == that.slot && this.type.equals(that.type);
  68     }
  69 
  70     @Override
  71     public int hashCode() {
  72         return super.hashCode();
  73     }
  74 
  75     @Override
  76     public String toString() {
  77         return "LocalImpl<name=" + name + ", type=" + type + ", startBci=" + startBci + ", endBci=" + endBci + ", slot=" + slot + ">";
  78     }
  79 }
