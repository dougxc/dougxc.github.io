  90 void CompilerToVM::Data::initialize(TRAPS) {
  91   Klass_vtable_start_offset = in_bytes(Klass::vtable_start_offset());
  92   Klass_vtable_length_offset = in_bytes(Klass::vtable_length_offset());
  93 
  94   Method_extra_stack_entries = Method::extra_stack_entries();
  95 
  96   SharedRuntime_ic_miss_stub = SharedRuntime::get_ic_miss_stub();
  97   SharedRuntime_handle_wrong_method_stub = SharedRuntime::get_handle_wrong_method_stub();
  98   SharedRuntime_deopt_blob_unpack = SharedRuntime::deopt_blob()->unpack();
  99   SharedRuntime_deopt_blob_uncommon_trap = SharedRuntime::deopt_blob()->uncommon_trap();
 100 
 101   ThreadLocalAllocBuffer_alignment_reserve = ThreadLocalAllocBuffer::alignment_reserve();
 102 
 103   Universe_collectedHeap = Universe::heap();
 104   Universe_base_vtable_size = Universe::base_vtable_size();
 105   Universe_narrow_oop_base = Universe::narrow_oop_base();
 106   Universe_narrow_oop_shift = Universe::narrow_oop_shift();
 107   Universe_narrow_klass_base = Universe::narrow_klass_base();
 108   Universe_narrow_klass_shift = Universe::narrow_klass_shift();
 109   Universe_non_oop_bits = Universe::non_oop_word();
 110   Universe_verify_oop_mask = Universe::verify_oop_mask();
 111   Universe_verify_oop_bits = Universe::verify_oop_bits();
 112 
 113   _supports_inline_contig_alloc = Universe::heap()->supports_inline_contig_alloc();
 114   _heap_end_addr = _supports_inline_contig_alloc ? Universe::heap()->end_addr() : (HeapWord**) -1;
 115   _heap_top_addr = _supports_inline_contig_alloc ? Universe::heap()->top_addr() : (HeapWord* volatile*) -1;
 116 
 117   _max_oop_map_stack_offset = (OopMapValue::register_mask - VMRegImpl::stack2reg(0)->value()) * VMRegImpl::stack_slot_size;
 118   int max_oop_map_stack_index = _max_oop_map_stack_offset / VMRegImpl::stack_slot_size;
 119   assert(OopMapValue::legal_vm_reg_name(VMRegImpl::stack2reg(max_oop_map_stack_index)), "should be valid");
 120   assert(!OopMapValue::legal_vm_reg_name(VMRegImpl::stack2reg(max_oop_map_stack_index + 1)), "should be invalid");
 121 
 122   symbol_init = (address) vmSymbols::object_initializer_name();
 123   symbol_clinit = (address) vmSymbols::class_initializer_name();
 124 
 125   BarrierSet* bs = BarrierSet::barrier_set();
 126   if (bs->is_a(BarrierSet::CardTableBarrierSet)) {
 127     jbyte* base = ci_card_table_address();
 128     assert(base != NULL, "unexpected byte_map_base");
 129     cardtable_start_address = base;
 130     cardtable_shift = CardTable::card_shift;
 131   } else {
 132     // No card mark barriers
 133     cardtable_start_address = 0;
 134     cardtable_shift = 0;
 135   }
 136 
 137   vm_page_size = os::vm_page_size();
 138 
 139 #define SET_TRIGFUNC(name)                                      \
 140   if (StubRoutines::name() != NULL) {                           \
 141     name = StubRoutines::name();                                \
 142   } else {                                                      \
 143     name = CAST_FROM_FN_PTR(address, SharedRuntime::name);      \
 144   }
 145 
 146   SET_TRIGFUNC(dsin);
 147   SET_TRIGFUNC(dcos);
 148   SET_TRIGFUNC(dtan);
 149   SET_TRIGFUNC(dexp);
 150   SET_TRIGFUNC(dlog10);
 151   SET_TRIGFUNC(dlog);
 152   SET_TRIGFUNC(dpow);
 153 
 154 #undef SET_TRIGFUNC
 155 }

 157 objArrayHandle CompilerToVM::initialize_intrinsics(TRAPS) {
 158   objArrayHandle vmIntrinsics = oopFactory::new_objArray_handle(VMIntrinsicMethod::klass(), (vmIntrinsics::ID_LIMIT - 1), CHECK_(objArrayHandle()));
 159   int index = 0;
 160   // The intrinsics for a class are usually adjacent to each other.
 161   // When they are, the string for the class name can be reused.
 162   vmSymbols::SID kls_sid = vmSymbols::NO_SID;
 163   Handle kls_str;
 164 #define VM_SYMBOL_TO_STRING(s) \
 165   java_lang_String::create_from_symbol(vmSymbols::symbol_at(vmSymbols::VM_SYMBOL_ENUM_NAME(s)), CHECK_(objArrayHandle()))
 166 #define VM_INTRINSIC_INFO(id, kls, name, sig, ignore_fcode) {             \
 167     instanceHandle vmIntrinsicMethod = InstanceKlass::cast(VMIntrinsicMethod::klass())->allocate_instance_handle(CHECK_(objArrayHandle())); \
 168     vmSymbols::SID sid = vmSymbols::VM_SYMBOL_ENUM_NAME(kls);             \
 169     if (kls_sid != sid) {                                                 \
 170       kls_str = VM_SYMBOL_TO_STRING(kls);                                 \
 171       kls_sid = sid;                                                      \
 172     }                                                                     \
 173     Handle name_str = VM_SYMBOL_TO_STRING(name);                          \
 174     Handle sig_str = VM_SYMBOL_TO_STRING(sig);                            \
 175     VMIntrinsicMethod::set_declaringClass(vmIntrinsicMethod, kls_str());  \
 176     VMIntrinsicMethod::set_name(vmIntrinsicMethod, name_str());           \
 177     VMIntrinsicMethod::set_descriptor(vmIntrinsicMethod, sig_str());      \
 178     VMIntrinsicMethod::set_id(vmIntrinsicMethod, vmIntrinsics::id);       \
 179       vmIntrinsics->obj_at_put(index++, vmIntrinsicMethod());             \
 180   }
 181 
 182   VM_INTRINSICS_DO(VM_INTRINSIC_INFO, VM_SYMBOL_IGNORE, VM_SYMBOL_IGNORE, VM_SYMBOL_IGNORE, VM_ALIAS_IGNORE)
 183 #undef VM_SYMBOL_TO_STRING
 184 #undef VM_INTRINSIC_INFO
 185   assert(index == vmIntrinsics::ID_LIMIT - 1, "must be");
 186 
 187   return vmIntrinsics;
 188 }

 255 #define BOXED_BOOLEAN(name, value) oop name = ((jboolean)(value) ? boxedTrue() : boxedFalse())
 256 #define BOXED_DOUBLE(name, value) oop name; do { jvalue p; p.d = (jdouble) (value); name = java_lang_boxing_object::create(T_DOUBLE, &p, CHECK_NULL);} while(0)
 257 #define BOXED_LONG(name, value) \
 258   oop name; \
 259   do { \
 260     jvalue p; p.j = (jlong) (value); \
 261     Handle* e = longs.get(p.j); \
 262     if (e == NULL) { \
 263       oop o = java_lang_boxing_object::create(T_LONG, &p, CHECK_NULL); \
 264       Handle h(THREAD, o); \
 265       longs.put(p.j, h); \
 266       name = h(); \
 267     } else { \
 268       name = (*e)(); \
 269     } \
 270   } while (0)
 271 
 272 #define CSTRING_TO_JSTRING(name, value) \
 273   Handle name; \
 274   do { \
 275     if (value != NULL) { \
 276       Handle* e = strings.get(value); \
 277       if (e == NULL) { \
 278         Handle h = java_lang_String::create_from_str(value, CHECK_NULL); \
 279         strings.put(value, h); \
 280         name = h; \
 281       } else { \
 282         name = (*e); \
 283       } \
 284     } \
 285   } while (0)

 287 jobjectArray readConfiguration0(JNIEnv *env, TRAPS) {
 288   ResourceMark rm;
 289   HandleMark hm;
 290 
 291   // Used to canonicalize Long and String values.
 292   ResourceHashtable<jlong, Handle> longs;
 293   ResourceHashtable<const char*, Handle, &CompilerToVM::cstring_hash, &CompilerToVM::cstring_equals> strings;
 294 
 295   jvalue prim;
 296   prim.z = true;  oop boxedTrueOop =  java_lang_boxing_object::create(T_BOOLEAN, &prim, CHECK_NULL);
 297   Handle boxedTrue(THREAD, boxedTrueOop);
 298   prim.z = false; oop boxedFalseOop = java_lang_boxing_object::create(T_BOOLEAN, &prim, CHECK_NULL);
 299   Handle boxedFalse(THREAD, boxedFalseOop);
 300 
 301   CompilerToVM::Data::initialize(CHECK_NULL);
 302 
 303   VMField::klass()->initialize(CHECK_NULL);
 304   VMFlag::klass()->initialize(CHECK_NULL);
 305   VMIntrinsicMethod::klass()->initialize(CHECK_NULL);
 306 
 307   int len = JVMCIVMStructs::localHotSpotVMStructs_count();
 308   objArrayHandle vmFields = oopFactory::new_objArray_handle(VMField::klass(), len, CHECK_NULL);
 309   for (int i = 0; i < len ; i++) {
 310     VMStructEntry vmField = JVMCIVMStructs::localHotSpotVMStructs[i];
 311     instanceHandle vmFieldObj = InstanceKlass::cast(VMField::klass())->allocate_instance_handle(CHECK_NULL);
 312     size_t name_buf_len = strlen(vmField.typeName) + strlen(vmField.fieldName) + 2 /* "::" */;
 313     char* name_buf = NEW_RESOURCE_ARRAY_IN_THREAD(THREAD, char, name_buf_len + 1);
 314     sprintf(name_buf, "%s::%s", vmField.typeName, vmField.fieldName);
 315     CSTRING_TO_JSTRING(name, name_buf);
 316     CSTRING_TO_JSTRING(type, vmField.typeString);
 317     VMField::set_name(vmFieldObj, name());
 318     VMField::set_type(vmFieldObj, type());
 319     VMField::set_offset(vmFieldObj, vmField.offset);
 320     VMField::set_address(vmFieldObj, (jlong) vmField.address);
 321     if (vmField.isStatic && vmField.typeString != NULL) {
 322       if (strcmp(vmField.typeString, "bool") == 0) {
 323         BOXED_BOOLEAN(box, *(jbyte*) vmField.address);
 324         VMField::set_value(vmFieldObj, box);
 325       } else if (strcmp(vmField.typeString, "int") == 0 ||
 326                  strcmp(vmField.typeString, "jint") == 0) {
 327         BOXED_LONG(box, *(jint*) vmField.address);
 328         VMField::set_value(vmFieldObj, box);
 329       } else if (strcmp(vmField.typeString, "uint64_t") == 0) {
 330         BOXED_LONG(box, *(uint64_t*) vmField.address);
 331         VMField::set_value(vmFieldObj, box);
 332       } else if (strcmp(vmField.typeString, "address") == 0 ||
 333                  strcmp(vmField.typeString, "intptr_t") == 0 ||
 334                  strcmp(vmField.typeString, "uintptr_t") == 0 ||
 335                  strcmp(vmField.typeString, "OopHandle") == 0 ||
 336                  strcmp(vmField.typeString, "size_t") == 0 ||
 337                  // All foo* types are addresses.
 338                  vmField.typeString[strlen(vmField.typeString) - 1] == '*') {
 339         BOXED_LONG(box, *((address*) vmField.address));
 340         VMField::set_value(vmFieldObj, box);
 341       } else {
 342         JVMCI_ERROR_NULL("VM field %s has unsupported type %s", name_buf, vmField.typeString);
 343       }
 344     }
 345     vmFields->obj_at_put(i, vmFieldObj());
 346   }
 347 
 348   int ints_len = JVMCIVMStructs::localHotSpotVMIntConstants_count();
 349   int longs_len = JVMCIVMStructs::localHotSpotVMLongConstants_count();
 350   len = ints_len + longs_len;
 351   objArrayHandle vmConstants = oopFactory::new_objArray_handle(SystemDictionary::Object_klass(), len * 2, CHECK_NULL);
 352   int insert = 0;
 353   for (int i = 0; i < ints_len ; i++) {
 354     VMIntConstantEntry c = JVMCIVMStructs::localHotSpotVMIntConstants[i];
 355     CSTRING_TO_JSTRING(name, c.name);
 356     BOXED_LONG(value, c.value);
 357     vmConstants->obj_at_put(insert++, name());
 358     vmConstants->obj_at_put(insert++, value);
 359   }
 360   for (int i = 0; i < longs_len ; i++) {
 361     VMLongConstantEntry c = JVMCIVMStructs::localHotSpotVMLongConstants[i];
 362     CSTRING_TO_JSTRING(name, c.name);
 363     BOXED_LONG(value, c.value);
 364     vmConstants->obj_at_put(insert++, name());
 365     vmConstants->obj_at_put(insert++, value);
 366   }
 367   assert(insert == len * 2, "must be");
 368 
 369   len = JVMCIVMStructs::localHotSpotVMAddresses_count();
 370   objArrayHandle vmAddresses = oopFactory::new_objArray_handle(SystemDictionary::Object_klass(), len * 2, CHECK_NULL);
 371   for (int i = 0; i < len ; i++) {
 372     VMAddressEntry a = JVMCIVMStructs::localHotSpotVMAddresses[i];
 373     CSTRING_TO_JSTRING(name, a.name);
 374     BOXED_LONG(value, a.value);
 375     vmAddresses->obj_at_put(i * 2, name());
 376     vmAddresses->obj_at_put(i * 2 + 1, value);
 377   }
 378 
 379 #define COUNT_FLAG(ignore) +1
 380 #ifdef ASSERT
 381 #define CHECK_FLAG(type, name) { \
 382   JVMFlag* flag = JVMFlag::find_flag(#name, strlen(#name), /*allow_locked*/ true, /* return_flag */ true); \
 383   assert(flag != NULL, "No such flag named " #name); \
 384   assert(flag->is_##type(), "JVMFlag " #name " is not of type " #type); \
 385 }
 386 #else
 387 #define CHECK_FLAG(type, name)
 388 #endif
 389 
 390 #define ADD_FLAG(type, name, convert) { \
 391   CHECK_FLAG(type, name) \
 392   instanceHandle vmFlagObj = InstanceKlass::cast(VMFlag::klass())->allocate_instance_handle(CHECK_NULL); \
 393   CSTRING_TO_JSTRING(fname, #name); \
 394   CSTRING_TO_JSTRING(ftype, #type); \
 395   VMFlag::set_name(vmFlagObj, fname()); \
 396   VMFlag::set_type(vmFlagObj, ftype()); \
 397   convert(value, name); \
 398   VMFlag::set_value(vmFlagObj, value); \
 399   vmFlags->obj_at_put(i++, vmFlagObj()); \
 400 }
 401 #define ADD_BOOL_FLAG(name)  ADD_FLAG(bool, name, BOXED_BOOLEAN)
 402 #define ADD_INTX_FLAG(name)  ADD_FLAG(intx, name, BOXED_LONG)
 403 #define ADD_UINTX_FLAG(name) ADD_FLAG(uintx, name, BOXED_LONG)
 404 
 405   len = 0 + PREDEFINED_CONFIG_FLAGS(COUNT_FLAG, COUNT_FLAG, COUNT_FLAG);
 406   objArrayHandle vmFlags = oopFactory::new_objArray_handle(VMFlag::klass(), len, CHECK_NULL);
 407   int i = 0;
 408   PREDEFINED_CONFIG_FLAGS(ADD_BOOL_FLAG, ADD_INTX_FLAG, ADD_UINTX_FLAG)
 409 
 410   objArrayHandle vmIntrinsics = CompilerToVM::initialize_intrinsics(CHECK_NULL);
 411 
 412   objArrayOop data = oopFactory::new_objArray(SystemDictionary::Object_klass(), 5, CHECK_NULL);
 413   data->obj_at_put(0, vmFields());
 414   data->obj_at_put(1, vmConstants());
 415   data->obj_at_put(2, vmAddresses());
 416   data->obj_at_put(3, vmFlags());
 417   data->obj_at_put(4, vmIntrinsics());
 418 
 419   return (jobjectArray) JNIHandles::make_local(THREAD, data);
 420 #undef COUNT_FLAG
 421 #undef ADD_FLAG
 422 #undef ADD_BOOL_FLAG
 423 #undef ADD_INTX_FLAG
 424 #undef ADD_UINTX_FLAG
 425 #undef CHECK_FLAG
 426 }
