  23 package jdk.vm.ci.hotspot.events;
  24 
  25 import jdk.vm.ci.common.JVMCIError;
  26 
  27 /**
  28  * An empty implementation for {@link EventProvider}. This implementation is used when no logging is
  29  * requested.
  30  */
  31 public final class EmptyEventProvider implements EventProvider {
  32 
  33     public CompilationEvent newCompilationEvent() {
  34         return new EmptyCompilationEvent();
  35     }
  36 
  37     public static class EmptyCompilationEvent implements CompilationEvent {
  38         public void commit() {
  39             throw JVMCIError.shouldNotReachHere();
  40         }
  41 
  42         public boolean shouldWrite() {
  43             // Events of this class should never been written.
  44             return false;
  45         }
  46 
  47         public void begin() {
  48         }
  49 
  50         public void end() {
  51         }
  52 
  53         public void setMethod(String method) {
  54             throw JVMCIError.shouldNotReachHere();
  55         }
  56 
  57         public void setCompileId(int compileId) {
  58             throw JVMCIError.shouldNotReachHere();
  59         }
  60 
  61         public void setCompileLevel(int compileLevel) {
  62             throw JVMCIError.shouldNotReachHere();
  63         }
  64 
  65         public void setSucceeded(boolean succeeded) {
  66             throw JVMCIError.shouldNotReachHere();
  67         }
  68 
  69         public void setIsOsr(boolean isOsr) {
  70             throw JVMCIError.shouldNotReachHere();
  71         }
  72 
  73         public void setCodeSize(int codeSize) {
  74             throw JVMCIError.shouldNotReachHere();
  75         }
  76 
  77         public void setInlinedBytes(int inlinedBytes) {
  78             throw JVMCIError.shouldNotReachHere();
  79         }
  80     }
  81 
  82     public CompilerFailureEvent newCompilerFailureEvent() {
  83         return new EmptyCompilerFailureEvent();
  84     }
  85 
  86     public static class EmptyCompilerFailureEvent implements CompilerFailureEvent {
  87         public void commit() {
  88             throw JVMCIError.shouldNotReachHere();
  89         }
  90 
  91         public boolean shouldWrite() {
  92             // Events of this class should never been written.
  93             return false;
  94         }
  95 
  96         public void setCompileId(int compileId) {
  97             throw JVMCIError.shouldNotReachHere();
  98         }
  99 
 100         public void setMessage(String message) {
 101             throw JVMCIError.shouldNotReachHere();
 102         }
 103     }
 104 
 105 }
