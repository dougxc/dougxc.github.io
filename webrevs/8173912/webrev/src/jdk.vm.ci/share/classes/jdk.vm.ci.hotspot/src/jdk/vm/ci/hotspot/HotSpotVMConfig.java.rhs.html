 249     /**
 250      * Returns the symbol in the {@code vmSymbols} table at position {@code index} as a
 251      * {@link String}.
 252      *
 253      * @param index position in the symbol table
 254      * @return the symbol at position id
 255      */
 256     String symbolAt(int index) {
 257         HotSpotJVMCIRuntimeProvider runtime = runtime();
 258         assert vmSymbolsFirstSID <= index && index < vmSymbolsSIDLimit : "index " + index + " is out of bounds";
 259         assert symbolPointerSize == Unsafe.ADDRESS_SIZE : "the following address read is broken";
 260         int offset = index * symbolPointerSize;
 261         return runtime.getCompilerToVM().getSymbol(UNSAFE.getAddress(vmSymbolsSymbols + offset));
 262     }

 343     String getCodeInstallResultDescription(int codeInstallResult) {
 344         if (codeInstallResult == codeInstallResultOk) {
 345             return "ok";
 346         }
 347         if (codeInstallResult == codeInstallResultDependenciesFailed) {
 348             return "dependencies failed";
 349         }
 350         if (codeInstallResult == codeInstallResultDependenciesInvalid) {
 351             return "dependencies invalid";
 352         }
 353         if (codeInstallResult == codeInstallResultCacheFull) {
 354             return "code cache is full";
 355         }
 356         if (codeInstallResult == codeInstallResultCodeTooLarge) {
 357             return "code is too large";
 358         }
 359         assert false : codeInstallResult;
 360         return "unknown";
 361     }
