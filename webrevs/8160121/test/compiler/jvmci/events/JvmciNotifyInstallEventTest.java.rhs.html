  88 import jdk.vm.ci.hotspot.HotSpotResolvedJavaMethod;
  89 
  90 public class JvmciNotifyInstallEventTest implements HotSpotVMEventListener {
  91     private static final String METHOD_NAME = "testMethod";
  92     private static final boolean FAIL_ON_INIT = !Boolean.getBoolean(
  93             "compiler.jvmci.events.JvmciNotifyInstallEventTest.failoninit");
  94     private static volatile int gotInstallNotification = 0;
  95 
  96     public static void main(String args[]) {
  97         new JvmciNotifyInstallEventTest().runTest();
  98     }
  99 
 100     private void runTest() {
 101         if (gotInstallNotification != 0) {
 102             throw new Error("Got install notification before test actions");
 103         }
 104         HotSpotCodeCacheProvider codeCache;
 105         try {
 106             codeCache = (HotSpotCodeCacheProvider) HotSpotJVMCIRuntime.runtime()
 107                     .getHostJVMCIBackend().getCodeCache();
 108         } catch (InternalError ie) {
 109             if (FAIL_ON_INIT) {
 110                 throw new AssertionError(
 111                         "Got unexpected InternalError trying to get code cache",
 112                         ie);
 113             }
 114             // passed
 115             return;
 116         }
 117         Asserts.assertTrue(FAIL_ON_INIT,
 118                     "Haven't caught InternalError in negative case");
 119         Method testMethod;
 120         try {
 121             testMethod = SimpleClass.class.getDeclaredMethod(METHOD_NAME);
 122         } catch (NoSuchMethodException e) {
 123             throw new Error("TEST BUG: Can't find " + METHOD_NAME, e);
 124         }
 125         HotSpotResolvedJavaMethod method = CTVMUtilities
 126                 .getResolvedMethod(SimpleClass.class, testMethod);
 127         HotSpotCompiledCode compiledCode = new HotSpotCompiledCode(METHOD_NAME,
 128                 new byte[0], 0, new Site[0], new Assumption[0],
 129                 new ResolvedJavaMethod[]{method}, new Comment[0], new byte[0],
 130                 16, new DataPatch[0], false, 0, null);
 131         codeCache.installCode(method, compiledCode, /* installedCode = */ null,
 132                 /* speculationLog = */ null, /* isDefault = */ false);
 133         Asserts.assertEQ(gotInstallNotification, 1,
 134                 "Got unexpected event count after 1st install attempt");
 135         // since "empty" compilation result is ok, a second attempt should be ok
 136         codeCache.installCode(method, compiledCode, /* installedCode = */ null,
 137                 /* speculationLog = */ null, /* isDefault = */ false);
 138         Asserts.assertEQ(gotInstallNotification, 2,
 139                 "Got unexpected event count after 2nd install attempt");
 140         // and an incorrect cases
 141         Utils.runAndCheckException(() -> {
 142             codeCache.installCode(method, null, null, null, true);
 143         }, NullPointerException.class);
 144         Asserts.assertEQ(gotInstallNotification, 2,
 145                 "Got unexpected event count after 3rd install attempt");
 146         Utils.runAndCheckException(() -> {
 147             codeCache.installCode(null, null, null, null, true);
 148         }, NullPointerException.class);
 149         Asserts.assertEQ(gotInstallNotification, 2,
 150                 "Got unexpected event count after 4th install attempt");
 151     }
 152 
 153     @Override
 154     public void notifyInstall(HotSpotCodeCacheProvider hotSpotCodeCacheProvider,
 155             InstalledCode installedCode, CompiledCode compiledCode) {
 156         gotInstallNotification++;
 157     }
 158 }
