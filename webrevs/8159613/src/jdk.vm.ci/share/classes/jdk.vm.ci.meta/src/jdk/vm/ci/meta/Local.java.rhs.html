  23 package jdk.vm.ci.meta;
  24 
  25 /**
  26  * Describes the type and bytecode index range in which a local variable is live.
  27  */
  28 public class Local {
  29 
  30     private final String name;
  31     private final int startBci;
  32     private final int endBci;
  33     private final int slot;
  34     private final JavaType type;
  35 
  36     public Local(String name, JavaType type, int startBci, int endBci, int slot) {
  37         this.name = name;
  38         this.startBci = startBci;
  39         this.endBci = endBci;
  40         this.slot = slot;
  41         this.type = type;
  42     }
  43 
  44     public int getStartBCI() {
  45         return startBci;
  46     }
  47 
  48     public int getEndBCI() {
  49         return endBci;
  50     }
  51 
  52     public String getName() {
  53         return name;
  54     }
  55 
  56     public JavaType getType() {
  57         return type;
  58     }
  59 
  60     public int getSlot() {
  61         return slot;
  62     }
  63 
  64     @Override
  65     public boolean equals(Object obj) {
  66         if (!(obj instanceof Local)) {
  67             return false;
  68         }
  69         Local that = (Local) obj;
  70         return this.name.equals(that.name) && this.startBci == that.startBci && this.endBci == that.endBci && this.slot == that.slot && this.type.equals(that.type);
  71     }
  72 
  73     @Override
  74     public int hashCode() {
  75         return super.hashCode();
  76     }
  77 
  78     @Override
  79     public String toString() {
  80         return "LocalImpl<name=" + name + ", type=" + type + ", startBci=" + startBci + ", endBci=" + endBci + ", slot=" + slot + ">";
  81     }
  82 }
